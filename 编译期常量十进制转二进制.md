//定义二进制常量
#define BYTE_BIN(n) ( \
((0##n%0100000000 /010000000>0)<<7)| \
((0##n%010000000  /01000000 >0)<<6)| \
((0##n%01000000   /0100000  >0)<<5)| \
((0##n%0100000    /010000   >0)<<4)| \
((0##n%010000     /01000    >0)<<3)| \
((0##n%01000      /0100     >0)<<2)| \
((0##n%0100       /010      >0)<<1)| \
((0##n%010        /01       >0)<<0))
